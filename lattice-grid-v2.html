// ============================================================ // LATTICE GRID BOT — REAL CAPITAL // Graduated spread: L1 0.30%, L2 0.5%, L3 0.7% // On BUY fill → places REAL SELL above it (never below cost basis) // On SELL fill → counts win, places REAL BUY below it // Auto top-up if open orders drop below target // NEVER sells below cost basis — holds until price recovers // Real market prices, REAL ORDERS, REAL CAPITAL ($500 total) // ============================================================ const PAIRS = ['BTC_USDT', 'ETH_USDT', 'SOL_USDT']; const LEVELS = 3; const LEVEL_SPACINGS = [0.003, 0.005, 0.007]; // L1 0.30%, L2 0.5%, L3 0.7% const MIN_PROFIT_MARGIN = 0.006; // minimum 0.6% above cost basis before selling const REAL_START_CAPITAL = 500; const BASE_URL = 'https://api.crypto.com/exchange/v1'; // ── Signing (Crypto.com Exchange v1 HMAC-SHA256) ───────────── function paramsToString(obj) { if (obj === null || obj === undefined) return ''; if (Array.isArray(obj)) { return obj.map(o => (o !== null && typeof o === 'object') ? paramsToString(o) : String(o)).join(''); } if (typeof obj === 'object') { return Object.keys(obj).sort().reduce((acc, k) => { const v = obj[k]; const valStr = (v !== null && typeof v === 'object') ? paramsToString(v) : String(v); return acc + k + valStr; }, ''); } return String(obj); } async function hmacSha256Hex(message, secret) { const enc = new TextEncoder(); const key = await crypto.subtle.importKey( 'raw', enc.encode(secret), { name: 'HMAC', hash: 'SHA-256' }, false, ['sign'] ); const sigBuf = await crypto.subtle.sign('HMAC', key, enc.encode(message)); return [...new Uint8Array(sigBuf)].map(b => b.toString(16).padStart(2, '0')).join(''); } let reqIdCounter = 1; async function callApi(env, method, params = {}) { const apiKey = env.CRYPTOCOM_API_KEY; const apiSecret = env.CRYPTOCOM_API_SECRET; const id = Date.now() + (reqIdCounter++ % 1000); const nonce = Date.now(); const paramsStr = paramsToString(params); const sigPayload = method + id + apiKey + paramsStr + nonce; const sig = await hmacSha256Hex(sigPayload, apiSecret); const body = { id, method, api_key: apiKey, params, nonce, sig }; const res = await fetch(`${BASE_URL}/${method}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(body) }); const data = await res.json(); if (data.code !== 0) { throw new Error(`${method} error ${data.code}: ${data.message || JSON.stringify(data)}`); } return data.result; } // ── Instrument precision (cached) ──────────────────────────── let instrumentMetaCache = {}; async function getInstrumentMeta(symbol) { if (instrumentMetaCache[symbol]) return instrumentMetaCache[symbol]; try { const res = await fetch(`${BASE_URL}/public/get-instruments`); const d = await res.json(); const list = d?.result?.data || []; for (const inst of list) { const sym = inst.symbol || inst.instrument_name; instrumentMetaCache[sym] = { priceDecimals: inst.price_decimals ?? inst.quote_decimals ?? 2, qtyDecimals: inst.quantity_decimals ?? inst.qty_decimals ?? 6 }; } } catch (e) {} return instrumentMetaCache[symbol] || { priceDecimals: 2, qtyDecimals: 6 }; } // ── Real order placement / status ──────────────────────────── async function placeOrder(env, symbol, side, price, qty, meta) { const priceStr = price.toFixed(meta.priceDecimals); const qtyStr = qty.toFixed(meta.qtyDecimals); const result = await callApi(env, 'private/create-order', { instrument_name: symbol, side, type: 'LIMIT', price: priceStr, quantity: qtyStr, time_in_force: 'GOOD_TILL_CANCEL' }); return result.order_id; } async function getOrderDetail(env, orderId) { return await callApi(env, 'private/get-order-detail', { order_id: String(orderId) }); } async function cancelOrder(env, symbol, orderId) { try { return await callApi(env, 'private/cancel-order', { instrument_name: symbol, order_id: String(orderId) }); } catch (e) { return null; } } async function getTicker(symbol) { try { const res = await fetch(`${BASE_URL}/public/get-tickers?instrument_name=${symbol}`); const d = await res.json(); return d?.result?.data?.[0] || null; } catch (e) { return null; } } // ── Real account balance (actual Crypto.com Exchange data) ─── async function getRealBalance(env) { try { const result = await callApi(env, 'private/user-balance', {}); return result; } catch (e) { return { error: e.message }; } } // ── State ───────────────────────────────────────────────────── async function loadState(env) { try { const s = JSON.parse((await env.LATTICE_BOT.get('grid_v2_state')) || 'null'); if (s) return s; } catch (e) {} const fresh = { startedAt: new Date().toISOString(), cash: REAL_START_CAPITAL, assets: {}, openOrders: {}, realizedPnl: 0, tradeCount: 0, winCount: 0, trades: [] }; for (const s of PAIRS) { fresh.assets[s] = { qty: 0, avgCost: 0 }; fresh.openOrders[s] = []; } return fresh; } async function saveState(env, state, newLogs) { await env.LATTICE_BOT.put('grid_v2_state', JSON.stringify(state)); if (newLogs && newLogs.length) { let existing = []; try { existing = JSON.parse((await env.LATTICE_BOT.get('grid_v2_logs')) || '[]'); } catch (e) {} await env.LATTICE_BOT.put('grid_v2_logs', JSON.stringify([...existing, ...newLogs].slice(-150))); } } // ── Main loop ───────────────────────────────────────────────── async function runGridBot(env) { const ts = new Date().toISOString(); const logs = []; const log = msg => logs.push({ ts, msg }); const state = await loadState(env); const usdtPerPair = state.cash / PAIRS.length; let equity = state.cash; for (const symbol of PAIRS) { const ticker = await getTicker(symbol); if (!ticker) { log(`[${symbol}] ticker failed, skipping`); continue; } const price = parseFloat(ticker.a || ticker.k || ticker.b || '0'); if (!price) continue; if (!state.assets[symbol]) state.assets[symbol] = { qty: 0, avgCost: 0 }; if (!state.openOrders[symbol]) state.openOrders[symbol] = []; const asset = state.assets[symbol]; let orders = state.openOrders[symbol]; const meta = await getInstrumentMeta(symbol); // ── Check real order status ────────────────────────────── const stillOpen = []; for (const o of orders) { let detail; try { detail = await getOrderDetail(env, o.orderId); } catch (e) { log(`[${symbol}] status check failed for order ${o.orderId}: ${e.message}`); stillOpen.push(o); // retry next run continue; } const status = detail?.status; if (status === 'CANCELED' || status === 'REJECTED' || status === 'EXPIRED') { log(`[${symbol}] order ${o.orderId} ${status}, dropping (will re-place on top-up)`); continue; } if (status !== 'FILLED') { stillOpen.push(o); // ACTIVE / PENDING / unknown → keep tracking continue; } // ── FILLED ── const fillPrice = parseFloat(detail.avg_price || o.price); const fillQty = parseFloat(detail.cumulative_quantity || o.qty); if (o.side === 'BUY') { const newQty = asset.qty + fillQty; asset.avgCost = newQty > 0 ? ((asset.avgCost * asset.qty) + (fillPrice * fillQty)) / newQty : fillPrice; asset.qty = newQty; state.cash -= fillQty * fillPrice; const targetSell = fillPrice * (1 + LEVEL_SPACINGS[0]); const costFloor = asset.avgCost * (1 + MIN_PROFIT_MARGIN); const sellPrice = Math.max(targetSell, costFloor); try { const newOrderId = await placeOrder(env, symbol, 'SELL', sellPrice, fillQty, meta); stillOpen.push({ orderId: newOrderId, side: 'SELL', price: sellPrice, qty: fillQty }); log(`[${symbol}] FILLED BUY @$${fillPrice.toFixed(2)} qty:${fillQty.toFixed(6)} → real SELL @$${sellPrice.toFixed(2)} (order ${newOrderId})`); } catch (e) { log(`[${symbol}] FILLED BUY but failed to place SELL: ${e.message}`); } } else { const profit = fillQty * (fillPrice - (asset.avgCost || fillPrice)); asset.qty = Math.max(0, asset.qty - fillQty); state.cash += fillQty * fillPrice; state.realizedPnl += profit; state.tradeCount++; if (profit > 0) state.winCount++; state.trades.unshift({ symbol, price: fillPrice, qty: fillQty, profit, ts }); state.trades = state.trades.slice(0, 20); const buyPrice = fillPrice * (1 - LEVEL_SPACINGS[0]); try { const newOrderId = await placeOrder(env, symbol, 'BUY', buyPrice, fillQty, meta); stillOpen.push({ orderId: newOrderId, side: 'BUY', price: buyPrice, qty: fillQty }); log(`[${symbol}] FILLED SELL @$${fillPrice.toFixed(2)} P&L:${profit >= 0 ? '+' : ''}$${profit.toFixed(4)} → real BUY @$${buyPrice.toFixed(2)} (order ${newOrderId})`); } catch (e) { log(`[${symbol}] FILLED SELL but failed to place BUY: ${e.message}`); } } } orders = stillOpen; // ── Top-up grid (real orders) ──────────────────────────── const buys = orders.filter(o => o.side === 'BUY'); const sells = orders.filter(o => o.side === 'SELL'); const perOrderUSD = Math.max(5, usdtPerPair / (LEVELS * 2)); if (buys.length < LEVELS && state.cash >= perOrderUSD) { for (let i = buys.length + 1; i <= LEVELS; i++) { const buyPrice = price * (1 - LEVEL_SPACINGS[i - 1]); const qty = perOrderUSD / buyPrice; if (qty > 0) { try { const orderId = await placeOrder(env, symbol, 'BUY', buyPrice, qty, meta); orders.push({ orderId, side: 'BUY', price: buyPrice, qty }); log(`[${symbol}] Top-up BUY L${i} @$${buyPrice.toFixed(2)} (order ${orderId})`); } catch (e) { log(`[${symbol}] Top-up BUY L${i} failed: ${e.message}`); } } } } if (sells.length < LEVELS && asset.qty > 0.0005) { const sellQtyEach = Math.max(0.0005, asset.qty / LEVELS); for (let i = sells.length + 1; i <= LEVELS; i++) { const targetSell = price * (1 + LEVEL_SPACINGS[i - 1]); const costFloor = (asset.avgCost > 0) ? asset.avgCost * (1 + MIN_PROFIT_MARGIN) : 0; const sellPrice = Math.max(targetSell, costFloor); try { const orderId = await placeOrder(env, symbol, 'SELL', sellPrice, sellQtyEach, meta); orders.push({ orderId, side: 'SELL', price: sellPrice, qty: sellQtyEach }); log(`[${symbol}] Top-up SELL L${i} @$${sellPrice.toFixed(2)}${costFloor > targetSell ? ' (held at cost floor)' : ''} (order ${orderId})`); } catch (e) { log(`[${symbol}] Top-up SELL L${i} failed: ${e.message}`); } } } state.openOrders[symbol] = orders; equity += asset.qty * price; } // ── Save ───────────────────────────────────────────────── state.equity = equity; state.pnlPct = ((equity - REAL_START_CAPITAL) / REAL_START_CAPITAL) * 100; state.winRate = state.tradeCount > 0 ? Math.round((state.winCount / state.tradeCount) * 100) : 0; state.lastRun = ts; await saveState(env, state, logs); return state; } // ── CORS ───────────────────────────────────────────────────── const CORS = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Methods': 'GET, POST, OPTIONS', 'Access-Control-Allow-Headers': 'Content-Type', 'Cache-Control': 'no-cache, no-store' }; const json = (d, s = 200) => new Response(JSON.stringify(d, null, 2), { status: s, headers: { ...CORS, 'Content-Type': 'application/json' } }); // ── Worker entry ───────────────────────────────────────────── export default { async scheduled(event, env, ctx) { ctx.waitUntil(runGridBot(env)); }, async fetch(request, env, ctx) { const path = new URL(request.url).pathname; if (request.method === 'OPTIONS') return new Response(null, { headers: CORS }); if (path === '/paper-grid-status') { const state = await loadState(env); let logs = []; try { logs = JSON.parse((await env.LATTICE_BOT.get('grid_v2_logs')) || '[]'); } catch (e) {} const realBalance = await getRealBalance(env); return json({ paperGrid: state, logs: logs.slice(-50), realBalance }); } if (path === '/paper-grid-run' && request.method === 'POST') { return json({ ok: true, paperGrid: await runGridBot(env) }); } if (path === '/paper-grid-reset' && request.method === 'POST') { // Cancel any live real orders before wiping tracked state try { const old = await loadState(env); for (const symbol of PAIRS) { for (const o of (old.openOrders?.[symbol] || [])) { await cancelOrder(env, symbol, o.orderId); } } } catch (e) {} const fresh = { startedAt: new Date().toISOString(), cash: REAL_START_CAPITAL, assets: {}, openOrders: {}, realizedPnl: 0, tradeCount: 0, winCount: 0, trades: [] }; for (const p of PAIRS) { fresh.assets[p] = { qty: 0, avgCost: 0 }; fresh.openOrders[p] = []; } await env.LATTICE_BOT.put('grid_v2_state', JSON.stringify(fresh)); await env.LATTICE_BOT.put('grid_v2_logs', JSON.stringify([])); return json({ ok: true, msg: 'Grid bot reset to $' + REAL_START_CAPITAL + ' (real orders cancelled)' }); } return json({ name: 'Lattice Grid Bot (REAL)', endpoints: ['/paper-grid-status', 'POST /paper-grid-run', 'POST /paper-grid-reset'] }); } };
